let soi_spawn_data: Option<(std::path::PathBuf, i64)> = session.history_db.as_ref()
    .and_then(|db| session.last_command_id.map(|cmd_id| (db.path().to_path_buf(), cmd_id)));

Then AFTER the entire `if let Some(ref db)` block AND after the snapshot/watcher blocks (around line ~2810, still inside the CommandFinished scope, before the git query section), spawn the worker:

```rust
// Spawn SOI parse worker thread
if let Some((db_path, cmd_id)) = soi_spawn_data {
    let proxy = self.proxy.clone();
    let wid = window_id;
    let sid = session_id;
    std::thread::Builder::new()
        .name("Glass SOI parse".into())
        .spawn(move || {
            let db = match glass_history::db::HistoryDb::open(&db_path) {
                Ok(db) => db,
                Err(e) => {
                    tracing::warn!("SOI worker: failed to open DB: {}", e);
                    return;
                }
            };

            let output_text = match db.get_output_for_command(cmd_id) {
                Ok(text) => text,
                Err(e) => {
                    tracing::warn!("SOI worker: failed to fetch output for cmd {}: {}", cmd_id, e);
                    None
                }
            };

            let command_text = db.get_command_text(cmd_id)
                .ok()
                .flatten()
                .unwrap_or_default();

            let (summary, severity) = match output_text {
                None => ("no output captured".to_string(), "Info".to_string()),
                Some(ref text) if text.is_empty() => {
                    ("no output captured".to_string(), "Info".to_string())
                }
                Some(text) => {
                    let output_type = glass_soi::classify(&text, Some(&command_text));
                    let parsed = glass_soi::parse(&text, output_type, Some(&command_text));
                    if let Err(e) = db.insert_parsed_output(cmd_id, &parsed) {
                        tracing::warn!("SOI: insert_parsed_output failed cmd={}: {}", cmd_id, e);
                    }
                    let sev_str = match parsed.summary.severity {
                        glass_soi::Severity::Error => "Error",
                        glass_soi::Severity::Warning => "Warning",
                        glass_soi::Severity::Info => "Info",
                        glass_soi::Severity::Success => "Success",
                    };
                    (parsed.summary.one_line, sev_str.to_string())
                }
            };

            let _ = proxy.send_event(AppEvent::SoiReady {
                window_id: wid,
                session_id: sid,
                command_id: cmd_id,
                summary,
                severity,
            });
        })
        .ok();
}
```

IMPORTANT placement details:
- The `soi_spawn_data` extraction MUST happen while `session` is still borrowed (inside the `if let Some(session)` block in the CommandFinished arm).
- The actual `std::thread::Builder::new().spawn()` can happen after the session borrow ends OR inside it -- `self.proxy.clone()` needs `self` access. Follow the same pattern as the git query spawn: extract needed data from session, then spawn outside the session borrow if needed.
- Check the borrow structure carefully. The git query spawn at line 2860 re-borrows session with `ctx.session_mux.session(session_id)` -- the SOI worker may need to do the same. Alternatively, capture `soi_spawn_data` early and spawn after the session borrow scope ends.

**Change 2: Handle AppEvent::SoiReady**

Add a new match arm in the `user_event` method, alongside the other AppEvent handlers (after `AppEvent::CommandOutput` or any convenient location):

```rust
AppEvent::SoiReady {
    window_id,
    session_id,
    command_id,
    summary,
    severity,
} => {
    if let Some(ctx) = self.windows.get_mut(&window_id) {
        if let Some(session) = ctx.session_mux.session_mut(session_id) {
            if session.last_command_id == Some(command_id) {
                session.last_soi_summary = Some(glass_mux::session::SoiSummary {
                    command_id,
                    one_line: summary,
                    severity,
                });
                tracing::debug!(
                    "SOI ready for cmd {}: {}",
                    command_id,
                    session.last_soi_summary.as_ref().unwrap().one_line
                );
            }
        }
        ctx.window.request_redraw();
    }
}
```

The `request_redraw()` call ensures downstream display (Phase 52) will pick up the new summary on next frame. For now it's a no-op visually since Phase 52 hasn't added rendering yet.
