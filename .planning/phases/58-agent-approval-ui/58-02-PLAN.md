**4. Keyboard shortcuts** in the `Ctrl+Shift` match arm (around line 1840-1960 where existing Ctrl+Shift+F, Ctrl+Shift+Z etc. are handled):

a) `Ctrl+Shift+A`: Toggle agent review overlay. Gate on `self.agent_runtime.is_some()`:
```rust
Key::Character(c) if c.as_str().eq_ignore_ascii_case("a") => {
    if self.agent_runtime.is_some() {

            self.proposal_review_selected = 0;

    }

```

b) `Ctrl+Shift+Y`: Accept selected proposal (only when overlay open):
```rust
Key::Character(c) if c.as_str().eq_ignore_ascii_case("y") && self.agent_review_open => {
    if !self.agent_proposal_worktrees.is_empty() {
        let idx = self.proposal_review_selected.min(self.agent_proposal_worktrees.len() - 1);
        let (_proposal, handle_opt) = self.agent_proposal_worktrees.remove(idx);
        if let (Some(wm), Some(handle)) = (self.worktree_manager.as_ref(), handle_opt) {
            if let Err(e) = wm.apply(handle) {
                tracing::error!("Failed to apply proposal: {e}");
            }
        }
        self.proposal_review_selected = self.proposal_review_selected.min(
            self.agent_proposal_worktrees.len().saturating_sub(1)
        );
        self.proposal_diff_cache = None;
        if self.agent_proposal_worktrees.is_empty() {
            self.agent_review_open = false;
        }
    }
    ctx.window.request_redraw();
    return;
}
```

c) `Ctrl+Shift+N`: Reject selected proposal (only when overlay open):
Same pattern as accept but call `wm.dismiss(handle)` instead.

d) Arrow keys when overlay open — these go OUTSIDE the Ctrl+Shift block. In the general keyboard handler, BEFORE the PTY forward, check:
```rust
if self.agent_review_open && event.state == ElementState::Pressed {
    match &event.logical_key {
        Key::Named(NamedKey::ArrowUp) => {
            self.proposal_review_selected = self.proposal_review_selected.saturating_sub(1);
            self.proposal_diff_cache = None;
            ctx.window.request_redraw();
            return;
        }
        Key::Named(NamedKey::ArrowDown) => {
            let max = self.agent_proposal_worktrees.len().saturating_sub(1);
            self.proposal_review_selected = (self.proposal_review_selected + 1).min(max);
            self.proposal_diff_cache = None;
            ctx.window.request_redraw();
            return;
        }
        Key::Named(NamedKey::Escape) => {
            self.agent_review_open = false;
            ctx.window.request_redraw();
            return;
        }
        _ => {} // CRITICAL: fall through to PTY — do NOT swallow (AGTU-05)
    }
}
```

c) Build overlay render data (with diff caching):
```rust
let proposal_overlay_data = if self.agent_review_open && !self.agent_proposal_worktrees.is_empty() {
    let selected = self.proposal_review_selected.min(self.agent_proposal_worktrees.len() - 1);
    // Generate diff on demand, cache it
    if self.proposal_diff_cache.as_ref().map_or(true, |(idx, _)| *idx != selected) {
        let diff = self.agent_proposal_worktrees.get(selected)
            .and_then(|(_, handle_opt)| handle_opt.as_ref())
            .and_then(|handle| self.worktree_manager.as_ref().map(|wm| wm.generate_diff(handle)))
            .and_then(|r| r.ok())
            .unwrap_or_default();
        self.proposal_diff_cache = Some((selected, diff));
    }
    let diff_preview = self.proposal_diff_cache.as_ref()
        .map(|(_, d)| d.clone())
        .unwrap_or_default();
    Some(glass_renderer::ProposalOverlayRenderData {
        proposals: self.agent_proposal_worktrees.iter()
            .map(|(p, _)| (p.description.clone(), p.action.clone()))
            .collect(),
        selected,
        diff_preview,
    })
} else {
    None
};
```
