```rust
// After block_manager.handle_event(), read temp files for pipeline stages
if let ShellEvent::PipelineStage { index, total_bytes, ref temp_path } = shell_event {
    // Read the temp file on the main thread (file should be small, <10MB capped by shell)
    match std::fs::read(&temp_path) {
        Ok(raw_bytes) => {
            // Process through StageBuffer with default policy
            let mut stage_buf = glass_pipes::StageBuffer::new(glass_pipes::BufferPolicy::default());
            stage_buf.append(&raw_bytes);
            let finalized = stage_buf.finalize();

            // Update the CapturedStage in the current block with real data
            if let Some(block) = ctx.block_manager.current_block_mut() {
                if let Some(stage) = block.pipeline_stages.iter_mut().find(|s| s.index == *index) {
                    stage.data = finalized;
                    stage.temp_path = None; // mark as read
                }
            }

            // Clean up the temp file
            let _ = std::fs::remove_file(&temp_path);
        }
        Err(e) => {
            eprintln!("glass: failed to read pipeline stage {index} from {temp_path}: {e}");
        }
    }
}

// On PipelineStart, clean up any stale temp dir from previous commands
// (defensive cleanup -- shell should have cleaned up but may not have on crash)
```
