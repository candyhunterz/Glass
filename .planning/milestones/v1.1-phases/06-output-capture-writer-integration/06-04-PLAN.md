**3. In the `AppEvent::Shell` handler (around line 420), AFTER `ctx.block_manager.handle_event(&osc_event, line)` and BEFORE the redraw request:**

Add handling for CommandExecuted and CommandFinished events:

```rust
// Track wall-clock start time on CommandExecuted
if matches!(shell_event, ShellEvent::CommandExecuted) {
    ctx.command_started_wall = Some(SystemTime::now());
}

// Insert CommandRecord on CommandFinished
if let ShellEvent::CommandFinished { exit_code } = &shell_event {
    if let Some(ref db) = ctx.history_db {
        let now = SystemTime::now();
        let finished_epoch = now
            .duration_since(SystemTime::UNIX_EPOCH)
            .map(|d| d.as_secs() as i64)
            .unwrap_or(0);
        let started_epoch = ctx.command_started_wall
            .and_then(|t| t.duration_since(SystemTime::UNIX_EPOCH).ok())
            .map(|d| d.as_secs() as i64)
            .unwrap_or(finished_epoch);
        let duration_ms = ctx.command_started_wall
            .and_then(|t| now.duration_since(t).ok())
            .map(|d| d.as_millis() as i64)
            .unwrap_or(0);

        // Command text: use empty string for now.
        // Extracting from terminal grid requires locking and reading lines
        // between command_start_line and output_start_line -- deferred to
        // a future enhancement. The metadata (cwd, exit_code, timestamps)
        // is the high-value data for HIST-02.
        let command_text = String::new();

        let record = CommandRecord {
            id: None,
            command: command_text,
            cwd: ctx.status.cwd().to_string(),
            exit_code: *exit_code,
            started_at: started_epoch,
            finished_at: finished_epoch,
            duration_ms,
            output: None, // Output arrives later via CommandOutput event
        };

        match db.insert_command(&record) {
            Ok(id) => {
                ctx.last_command_id = Some(id);
                tracing::debug!("Inserted command record id={}", id);
            }
            Err(e) => {
                tracing::warn!("Failed to insert command record: {}", e);
                ctx.last_command_id = None;
            }
        }
    }
    // Clear wall-clock tracker
    ctx.command_started_wall = None;
}
```

**IMPORTANT NOTES:**
- Command text is empty string for now. Extracting it requires reading the terminal grid (locking the FairMutex) between command_start_line and output_start_line. This is a known limitation -- the metadata (cwd, exit_code, timestamps, output) is the primary value. Command text extraction can be added in Phase 7 or as a follow-up.
- HistoryDb::open failure is non-fatal -- history is best-effort, never crashes the terminal.
- SystemTime for wall-clock (not Instant) because CommandRecord needs Unix epoch seconds.
- CommandOutput arrives AFTER CommandFinished (OutputBuffer.finish() fires on CommandFinished, bytes go through AppEvent), so insert first, then update with output.
